useScrollLock

주로 모달과 같은 팝업이 뜰 때 스크롤을 이벤트를 막아주는 훅으로 사용

Hooks

tsx
1import { useCallback, useRef } from 'react';  
2  
3type CSSProperties = {  
4  [key: string]: string | number;  
5};  
6  
7export default function useScrollLock() {  
8  const scrollRef = useRef(0);  
9  const $body = document.querySelector('body');  
10  
11  const modifyBodyStyle = useCallback(  
12    (style: CSSProperties) => {  
13      if ($body) {  
14        Object.assign($body.style, style);  
15      } else {  
16        console.error('document.body is not defined');  
17      }  
18    },  
19    [$body],  
20  );  
21  
22  const lockScroll = () => {  
23    scrollRef.current = window.scrollY;  
24    modifyBodyStyle({  
25      overflowY: 'hidden',  
26      position: 'fixed',  
27      top: `-${scrollY}px`,  
28      left: '0',  
29      right: '0',  
30    });  
31  };  
32  
33  const unlockScroll = () => {  
34    modifyBodyStyle({  
35      overflowY: '',  
36      position: '',  
37      top: '',  
38      left: '',  
39      right: '',  
40    });  
41    window.scrollTo({ left: 0, top: scrollRef.current, behavior: 'instant' });  
42  };  
43  
44  return {  
45    lockScroll,  
46    unlockScroll,  
47  };  
48}
